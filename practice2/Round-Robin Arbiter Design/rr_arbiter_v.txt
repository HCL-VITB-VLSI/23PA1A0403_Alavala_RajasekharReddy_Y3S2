module round_robin_arbiter #(
    parameter WIDTH = 4
)(
    input  logic clk,
    input  logic rst,
    input  logic [WIDTH-1:0] req,
    output logic [WIDTH-1:0] grant
);

logic [$clog2(WIDTH)-1:0] ptr;

integer i;
integer idx;
integer found;

always_ff @(posedge clk or posedge rst) begin
    if (rst) begin
        ptr   <= 0;
        grant <= 0;
    end
    else begin
        grant <= 0;
        found = 0;

        for (i = 0; i < WIDTH; i = i + 1) begin
            idx = (ptr + i) % WIDTH;

            if (req[idx] && !found) begin
                grant[idx] <= 1'b1;
                ptr <= (idx + 1) % WIDTH;
                found = 1;
            end
        end
    end
end

endmodule
